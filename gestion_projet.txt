\section{Résumé Exécutif}
% ==========================================

\subsection{Vue d'ensemble du projet}
Flight Tracker Pro est une application web de surveillance du trafic aérien mondial en temps réel. Le projet vise à créer une solution gratuite, performante et complète pour visualiser et analyser les données de vol provenant de l'API OpenSky Network.

\subsection{Objectifs principaux}
\begin{itemize}[label=\textcolor{blue}{$\bullet$}]
    \item Développer une interface web interactive et moderne
    \item Intégrer l'API OpenSky Network avec authentification OAuth2
    \item Créer des visualisations cartographiques performantes
    \item Implémenter des analyses statistiques avancées
    \item Assurer une expérience utilisateur fluide et intuitive
\end{itemize}

\subsection{Résultats obtenus}
\begin{tcolorbox}[colback=green!5!white,colframe=green!75!black,title=Succès du Projet]
\begin{itemize}
    \item Application web complète avec 7 onglets fonctionnels
    \item Visualisation de 10 000+ vols simultanés en temps réel
    \item Performance optimale avec rafraîchissement < 3 secondes
    \item Reconnaissance automatique de 24+ compagnies aériennes
    \item Interface moderne en bleu et blanc
\end{itemize}
\end{tcolorbox}

\subsection{Phase 2 : Développement du Backend (Semaine 2)}

\subsubsection{Intégration API}
\begin{enumerate}
    \item Mise en place de l'authentification OAuth2
    \item Développement de la fonction \texttt{get\_oauth\_token()}
    \item Création de \texttt{fetch\_flights()} pour récupérer les données
    \item Gestion du cache avec \texttt{st.session\_state}
    \item Implémentation du mode anonyme de secours
\end{enumerate}

\subsubsection{Traitement des données}
\begin{itemize}
    \item Nettoyage et validation des données
    \item Conversion des unités (m/s → knots, m → ft)
    \item Calcul des statuts de vol (montée, descente, croisière, au sol)
    \item Extraction des compagnies aériennes depuis les callsigns
    \item Calcul des directions cardinales
\end{itemize}

\subsubsection{Défis rencontrés}
\begin{tcolorbox}[colback=red!5!white,colframe=red!75!black,title=Défis Techniques]
\begin{itemize}
    \item \textbf{Timeout API} : Résolu avec augmentation du timeout à 30s et mode anonyme
    \item \textbf{Données manquantes} : Gestion robuste des valeurs NaN
    \item \textbf{Performance} : Optimisation avec cache et traitement vectorisé
\end{itemize}
\end{tcolorbox}

\section{Difficultés Rencontrées et Solutions}
% ==========================================

\subsection{Problèmes techniques}

\subsubsection{1. Performance de la carte}
\textbf{Problème :} La carte initiale avec Plotly Scattergeo était très lente et laggait avec des milliers de points.

\textbf{Solution :}
\begin{itemize}
    \item Migration vers Scattermapbox avec Mapbox GL
    \item Hardware acceleration activée
    \item Marqueurs simplifiés (cercles au lieu de symboles complexes)
    \item Résultat : Performances similaires à FlightRadar24
\end{itemize}

\subsubsection{2. Marqueurs invisibles}
\textbf{Problème :} Tentatives d'utiliser des icônes d'avion (SVG, emojis, triangles) mais rien n'apparaissait sur la carte.

\textbf{Solution :}
\begin{itemize}
    \item Abandon des icônes complexes
    \item Utilisation de simples cercles colorés (mode='markers')
    \item Couleurs distinctes par statut de vol
    \item Taille réduite à 5px pour esthétique
\end{itemize}

\subsubsection{3. Timeout d'authentification}
\textbf{Problème :} Connexion au serveur OpenSky parfois impossible (timeout).

\textbf{Solution :}
\begin{itemize}
    \item Augmentation du timeout de 10s à 30s
    \item Implémentation d'un mode anonyme de secours
    \item Message d'avertissement à l'utilisateur
    \item Gestion gracieuse des erreurs
\end{itemize}

\subsubsection{4. Conflits de widgets Streamlit}
\textbf{Problème :} Deux selectbox avec la même variable causaient des bugs.

\textbf{Solution :}
\begin{itemize}
    \item Renommage des variables (selected\_airline → selected\_airline\_detail)
    \item Ajout de keys uniques pour chaque widget
    \item Meilleure organisation du code
\end{itemize}

\subsection{Défis organisationnels}

\begin{itemize}
    \item \textbf{Communication} : Réunions quotidiennes courtes (stand-ups)
    \item \textbf{Scope creep} : Priorisation stricte des fonctionnalités
    \item \textbf{Documentation} : Documentation continue pendant le développement
\end{itemize}

\subsection{Leçons apprises}

\begin{tcolorbox}[colback=blue!5!white,colframe=blue!75!black,title=Leçons Clés]
\begin{enumerate}
    \item \textbf{Simplicité d'abord} : Les solutions simples (cercles) fonctionnent mieux que les complexes (SVG)
    \item \textbf{Performance critique} : L'UX dépend fortement de la fluidité
    \item \textbf{Gestion d'erreurs} : Toujours prévoir un mode dégradé
    \item \textbf{Feedback utilisateur} : Les retours en temps réel améliorent le produit
    \item \textbf{Documentation continue} : Documenter au fur et à mesure, pas à la fin
\end{enumerate}
\end{tcolorbox}
